#Add to scripted triggers if more levels of hyper inflation is added


#Important to put HIGHEST first
MAL_hidden_cost_reduction_up = {
	if = {
		limit = { has_idea = MAL_hidden_cost_reduction_5 }
		swap_ideas = {
			remove_idea = MAL_hidden_cost_reduction_5
			add_idea = MAL_hidden_cost_reduction_6
		}
	}
	else_if = {
		limit = { has_idea = MAL_hidden_cost_reduction_4 }
		swap_ideas = {
			remove_idea = MAL_hidden_cost_reduction_4
			add_idea = MAL_hidden_cost_reduction_5
		}
	}
	else_if = {
		limit = { has_idea = MAL_hidden_cost_reduction_3 }
		swap_ideas = {
			remove_idea = MAL_hidden_cost_reduction_3
			add_idea = MAL_hidden_cost_reduction_4
		}
	}
	else_if = {
		limit = { has_idea = MAL_hidden_cost_reduction_2 }
		swap_ideas = {
			remove_idea = MAL_hidden_cost_reduction_2
			add_idea = MAL_hidden_cost_reduction_3
		}
	}
	else_if = {
		limit = { has_idea = MAL_hidden_cost_reduction }
		swap_ideas = {
			remove_idea = MAL_hidden_cost_reduction
			add_idea = MAL_hidden_cost_reduction_2
		}		
	}
	else_if = {
		limit = { NOT = { has_idea = MAL_hidden_cost_reduction } }
		add_ideas = MAL_hidden_cost_reduction
	}
}
